#include <iostream>

i++
int toplama(int a, int b) {
    return a + b;
}

int cikarma(int a, int b) {
    return a - b;
}

int carpma(int a, int b) {
    return a * b;
}

float bolme(float a, float b) {
    if (b != 0) {
        return a / b;
    } else {
        std::cout << "Hata" << std::endl;
        return 0;
    }
}

void selamla() {
    std::cout << "Merhaba!" << std::endl;
}


int main() {
    int x = 5;
    int y = 3;
    
    int sonuc1 = toplama(x, y);
    int sonuc2 = cikarma(x, y);
    int sonuc3 = carpma(x, y);
    float sonuc4 = bolme(x, y);
    
    std::cout << "Toplama: " << sonuc1 << std::endl;
    std::cout << " " << sonuc2 << std::endl;
    std::cout << " " << sonuc3 << std::endl;
    std::cout << " " << sonuc4 << std::endl;
    
    selamla();
    
    
    return 0;
}
